# v2.1.0 Stage CLIs Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add the seven decoupled stage CLIs — `i18n`, `get`, `set`, `history`, `diff`, `restore`, `refresh` — so users can run individual stages on top of an existing `out/<slug>/record.json` without re-crawling.

**Architecture:** A subcommand dispatcher at the top of `framework/cli.mjs` routes `argv[0]` to one of seven thin modules under `framework/commands/<name>.mjs`. Each module is a default-export async function that takes `(args, context)` and does its job: read/write `out/<slug>/`, validate via the existing schema validator, commit via `version-store`, regenerate `out/index.html` via `buildOutBrowser`. No new infrastructure — every command composes existing v2.0 primitives.

**Tech Stack:** Node.js (no new deps), `framework/version-store.mjs` (v2.0), `framework/schema-validator.mjs`, `framework/subtask-runner.mjs`, `framework/merger.mjs::mergeR2`, `framework/i18n-stage.mjs`, `framework/out-browser.mjs::buildOutBrowser`. Zero-dep test runner at `tests/run.mjs`.

**Out of scope (deferred to v2.2.0):** Slash command surface for the stage CLIs (`/protocol-info:i18n pendle` etc.). Users in v2.1 invoke via `bash run.sh i18n pendle` directly.

**Spec:** `docs/superpowers/specs/2026-04-27-decouple-stages-and-versioning-design.md` (commit `a16425d`).

**Builds on v2.0:** `framework/version-store.mjs`, flat `out/<slug>/` layout, refuse-clobber invariant.

## Architecture diagrams

### Subcommand dispatcher flow

```
  bash run.sh i18n pendle --locales zh_CN,ja_JP
                         │
                         ▼
            ┌─────────────────────────────────┐
            │  framework/cli.mjs entry        │
            │  argv = ['i18n','pendle',...]   │
            └──────────────┬──────────────────┘
                           │
                           ▼
            ┌─────────────────────────────────┐
            │  if argv[0] in STAGE_COMMANDS:  │
            │    dynamic import cli/<name>.mjs│
            │    invoke default(argv.slice(1))│
            │  else:                          │
            │    fall through to crawl logic  │ ◄── existing v2.0 path
            └──────────────┬──────────────────┘
                           │
                           ▼
            ┌─────────────────────────────────┐
            │  cli/i18n.mjs default export    │
            │   load out/pendle/record.json   │
            │   run i18n-stage                │
            │   write out/pendle/record.full  │
            │   commit + regen index.html     │
            └─────────────────────────────────┘
```

### Stage CLI commit pattern (i18n / set / restore / refresh)

```
load out/<slug>/record.json (must exist; else exit 1)
        │
        ▼
do work (translate / set path / checkout / re-run subtask)
        │
        ▼
optional: validate result against schema (set, restore, refresh)
        │
        ▼
write modified file(s) under out/<slug>/
        │
        ▼
commit(out, { paths: [<slug>/...], message: "<stage>(<slug>) ...", runId })
        │ runId is a fresh timestamp generated by the CLI itself
        │ (each stage CLI invocation = one logical "run")
        ▼
buildOutBrowser(out)  ◄── refresh out/index.html so the browser stays current
```

### `get` / `history` / `diff` (read-only) pattern

```
parse args → call version-store / read record.json → print to stdout
                                                   → exit 0 (or 1 on miss)
NO commit. NO browser regen.
```

// ── Subcommand dispatch (v2.1) ──────────────────────────────────────────────
// Stage CLIs live in framework/commands/<name>.mjs as default-export async fns.
// The map values are lazy importers so we don't load every module on every run.

export const STAGE_COMMANDS = {
  i18n:    () => import('./commands/i18n.mjs'),
  get:     () => import('./commands/get.mjs'),
  set:     () => import('./commands/set.mjs'),
  history: () => import('./commands/history.mjs'),
  diff:    () => import('./commands/diff.mjs'),
  restore: () => import('./commands/restore.mjs'),
  refresh: () => import('./commands/refresh.mjs'),
};

export async function dispatch(argv, { commandMap = STAGE_COMMANDS, crawl = null } = {}) {
  const sub = argv[0];
  if (sub && Object.prototype.hasOwnProperty.call(commandMap, sub)) {
    const mod = await commandMap[sub]();
    const fn = mod.default;
    if (typeof fn !== 'function') {
      throw new Error(`framework/commands/${sub}.mjs must default-export a function`);
    }
    return await fn(argv.slice(1));
  }
  // Fall through to crawl (existing v2.0 behavior).
  if (typeof crawl === 'function') return await crawl(argv);
  return runCrawl(argv); // existing logic, see below
}
```

Then refactor the existing argv-parsing + `run()` invocation block into a top-level `async function runCrawl(argv) { ... }` that takes the argv list. The existing main block becomes:

// framework/jsonpath.mjs — tiny JSONPath dialect: a.b[0].c. No filters, no
// recursive descent, no slicing. Used by `protocol-info get` and `set`.

const SEGMENT = /^([a-zA-Z_][a-zA-Z0-9_]*)((?:\[\d+\])*)$/;

export function parse(path) {
  if (typeof path !== 'string' || path.length === 0) {
    throw new Error('jsonpath: empty path');
  }
  if (path.includes('..')) {
    throw new Error(`jsonpath: invalid path (consecutive dots): ${path}`);
  }
  const segments = [];
  for (const part of path.split('.')) {
    const m = SEGMENT.exec(part);
    if (!m) throw new Error(`jsonpath: unsupported or invalid segment: ${part}`);
    segments.push(m[1]);
    if (m[2]) {
      // m[2] looks like "[0]" or "[0][1]"
      for (const idx of m[2].matchAll(/\[(\d+)\]/g)) {
        segments.push(Number(idx[1]));
      }
    }
  }
  return segments;
}

export function getAt(obj, path) {
  const segs = parse(path);
  let cur = obj;
  for (const s of segs) {
    if (cur == null || typeof cur !== 'object') {
      throw new Error(`jsonpath: not found at "${path}" (parent is ${typeof cur})`);
    }
    if (typeof s === 'number') {
      if (!Array.isArray(cur)) throw new Error(`jsonpath: expected array at "${path}"`);
      if (s < 0 || s >= cur.length) throw new Error(`jsonpath: index ${s} out of bounds at "${path}"`);
      cur = cur[s];
    } else {
      if (!(s in cur)) throw new Error(`jsonpath: key "${s}" not found at "${path}"`);
      cur = cur[s];
    }
  }
  return cur;
}

export function setAt(obj, path, value) {
  const segs = parse(path);
  if (segs.length === 0) throw new Error('jsonpath: empty path');
  let cur = obj;
  for (let i = 0; i < segs.length - 1; i++) {
    const s = segs[i];
    if (cur == null || typeof cur !== 'object') {
      throw new Error(`jsonpath: not found at "${path}" (parent is ${typeof cur})`);
    }
    if (typeof s === 'number') {
      if (!Array.isArray(cur)) throw new Error(`jsonpath: expected array at "${path}"`);
      if (s < 0 || s >= cur.length) throw new Error(`jsonpath: index ${s} out of bounds at "${path}"`);
    } else {
      if (!(s in cur)) throw new Error(`jsonpath: key "${s}" not found at "${path}"`);
    }
    cur = cur[s];
  }
  const last = segs[segs.length - 1];
  if (typeof last === 'number') {
    if (!Array.isArray(cur)) throw new Error(`jsonpath: expected array at "${path}"`);
    if (last < 0 || last >= cur.length) throw new Error(`jsonpath: index ${last} out of bounds at "${path}"`);
  }
  cur[last] = value;
  return obj;
}
```

// framework/commands/i18n.mjs — re-translate an existing record without re-crawling.
//
// Usage: protocol-info i18n <slug> [--locales zh_CN,ja_JP|all]

import { existsSync } from 'node:fs';
import { writeFile } from 'node:fs/promises';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { spawn } from 'node:child_process';
import { commit } from '../version-store.mjs';
import { buildOutBrowser } from '../out-browser.mjs';
import { loadManifest } from '../manifest-loader.mjs';
import { resolveI18nSelection } from '../orchestrator.mjs';

const FRAMEWORK_DIR = dirname(fileURLToPath(import.meta.url));
const SCRIPT_DIR = dirname(dirname(FRAMEWORK_DIR));

function freshRunId() {
  return new Date().toISOString().replace(/[-:]/g, '').replace(/\.\d{3}Z$/, 'Z');
}

function defaultRunI18nStage({ slugDir, locales, manifestPath, model }) {
  return new Promise((resolve) => {
    const args = [
      join(FRAMEWORK_DIR, '..', 'cli', 'i18n.mjs'), // existing v2.0 i18n-stage CLI
      '--manifest', manifestPath,
      '--record', join(slugDir, 'record.json'),
      '--output-dir', join(slugDir, '_debug', 'i18n'),
      '--locales', locales.join(','),
    ];
    if (model) args.push('--model', model);
    const proc = spawn('node', args, { stdio: 'inherit' });
    proc.on('close', resolve);
  });
}

export default async function i18nCmd(args, ctx = {}) {
  const outputRoot = ctx.outputRoot || join(SCRIPT_DIR, 'out');
  const runI18nStage = ctx.runI18nStage || defaultRunI18nStage;

  const slug = args[0];
  if (!slug) {
    process.stderr.write('Usage: protocol-info i18n <slug> [--locales zh_CN,ja_JP|all]\n');
    return 1;
  }
  let localesArg = '';
  for (let i = 1; i < args.length; i++) {
    if (args[i] === '--locales') localesArg = args[++i] || '';
  }

  const slugDir = join(outputRoot, slug);
  if (!existsSync(join(slugDir, 'record.json'))) {
    process.stderr.write(`i18n: ${join(slugDir, 'record.json')} does not exist. Run 'crawl' first.\n`);
    return 1;
  }

  const manifestPath = ctx.manifestPath || join(SCRIPT_DIR, 'consumers', 'protocol-info', 'manifest.json');
  const manifest = await loadManifest(manifestPath);
  const locales = resolveI18nSelection(localesArg, manifest);
  if (locales.length === 0) {
    process.stderr.write(`i18n: no locales selected (got "${localesArg}")\n`);
    return 1;
  }

  const code = await runI18nStage({ slugDir, locales, manifestPath, model: ctx.i18nModel });
  if (code !== 0) {
    process.stderr.write(`i18n: stage exited ${code}\n`);
    return code;
  }

  const runId = freshRunId();
  await commit(outputRoot, {
    paths: [`${slug}/record.full.json`],

// framework/commands/set.mjs — edit one JSON path with schema validation + commit.
// Usage: protocol-info set <slug> <jsonpath> <json-value>

import { readFile, writeFile } from 'node:fs/promises';
import { existsSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { setAt } from '../jsonpath.mjs';
import { commit } from '../version-store.mjs';
import { buildOutBrowser } from '../out-browser.mjs';
import { validateRecord } from '../schema-validator.mjs';
import { loadManifest } from '../manifest-loader.mjs';

const FRAMEWORK_DIR = dirname(fileURLToPath(import.meta.url));
const SCRIPT_DIR = dirname(dirname(FRAMEWORK_DIR));

function freshRunId() {
  return new Date().toISOString().replace(/[-:]/g, '').replace(/\.\d{3}Z$/, 'Z');
}

async function defaultValidate(record, manifestPath) {
  const manifest = await loadManifest(manifestPath);
  const result = await validateRecord(record, manifest);
  return { ok: result.ok, errors: result.errors || [] };
}

export default async function setCmd(args, ctx = {}) {
  const outputRoot = ctx.outputRoot || join(SCRIPT_DIR, 'out');
  const manifestPath = ctx.manifestPath || join(SCRIPT_DIR, 'consumers', 'protocol-info', 'manifest.json');
  const validate = ctx.validate || ((rec) => defaultValidate(rec, manifestPath));

  const [slug, jsonpath, valueArg] = args;
  if (!slug || !jsonpath || valueArg === undefined) {
    process.stderr.write('Usage: protocol-info set <slug> <jsonpath> <json-value>\n');
    process.stderr.write('  <json-value> is parsed as JSON: strings need quotes, booleans/numbers/null bare.\n');
    return 1;
  }
  const recordPath = join(outputRoot, slug, 'record.json');
  if (!existsSync(recordPath)) {
    process.stderr.write(`set: ${recordPath} does not exist\n`);
    return 1;
  }
  let value;
  try {
    value = JSON.parse(valueArg);
  } catch (err) {
    process.stderr.write(`set: <json-value> is not valid JSON: ${err.message}\n`);
    return 1;
  }
  const record = JSON.parse(await readFile(recordPath, 'utf8'));
  try {
    setAt(record, jsonpath, value);
  } catch (err) {
    process.stderr.write(`set: ${err.message}\n`);
    return 1;
  }

  const result = await validate(record);
  if (!result.ok) {
    process.stderr.write(`set: validation failed (${result.errors.length} errors). Record NOT written.\n`);
    for (const e of result.errors.slice(0, 5)) {
      process.stderr.write(`  ${e.path || '/'}: ${e.message}\n`);
    }
    return 1;
  }

  await writeFile(recordPath, JSON.stringify(record, null, 2) + '\n');
  await commit(outputRoot, {
    paths: [`${slug}/record.json`],
    message: `set(${slug}) ${jsonpath}`,
    runId: freshRunId(),
  });
  await buildOutBrowser(outputRoot);
  return 0;
}

// framework/commands/restore.mjs — git checkout + revalidate + commit.
// Usage: protocol-info restore <slug> <sha>

import { readFile } from 'node:fs/promises';
import { existsSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { restore, commit } from '../version-store.mjs';
import { buildOutBrowser } from '../out-browser.mjs';
import { validateRecord } from '../schema-validator.mjs';
import { loadManifest } from '../manifest-loader.mjs';

const FRAMEWORK_DIR = dirname(fileURLToPath(import.meta.url));
const SCRIPT_DIR = dirname(dirname(FRAMEWORK_DIR));

function freshRunId() {
  return new Date().toISOString().replace(/[-:]/g, '').replace(/\.\d{3}Z$/, 'Z');
}

async function defaultValidate(record, manifestPath) {
  const manifest = await loadManifest(manifestPath);
  const result = await validateRecord(record, manifest);
  return { ok: result.ok, errors: result.errors || [] };
}

export default async function restoreCmd(args, ctx = {}) {
  const outputRoot = ctx.outputRoot || join(SCRIPT_DIR, 'out');
  const manifestPath = ctx.manifestPath || join(SCRIPT_DIR, 'consumers', 'protocol-info', 'manifest.json');
  const validate = ctx.validate || ((rec) => defaultValidate(rec, manifestPath));

  const [slug, sha] = args;
  if (!slug || !sha) {
    process.stderr.write('Usage: protocol-info restore <slug> <sha>\n');
    return 1;
  }
  await restore(outputRoot, { slug, sha });

  const recordPath = join(outputRoot, slug, 'record.json');
  if (!existsSync(recordPath)) {
    process.stderr.write(`restore: ${recordPath} missing after checkout — nothing to validate\n`);
    return 1;
  }
  const record = JSON.parse(await readFile(recordPath, 'utf8'));
  const result = await validate(record);
  if (!result.ok) {
    process.stderr.write(`restore: post-checkout validation failed (${result.errors.length} errors). NOT committing.\n`);
    for (const e of result.errors.slice(0, 5)) {
      process.stderr.write(`  ${e.path || '/'}: ${e.message}\n`);
    }
    return 1;
  }

  await commit(outputRoot, {
    paths: [`${slug}/`],
    message: `restore(${slug}) ${sha}`,
    runId: freshRunId(),
  });
  await buildOutBrowser(outputRoot);
  return 0;
}
```

// framework/commands/refresh.mjs — re-run one subtask, merge, commit.
// Usage: protocol-info refresh <slug> <subtask>

import { readFile, writeFile } from 'node:fs/promises';
import { existsSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { commit } from '../version-store.mjs';
import { buildOutBrowser } from '../out-browser.mjs';
import { mergeR2 } from '../merger.mjs';
import { validateRecord } from '../schema-validator.mjs';
import { loadManifest } from '../manifest-loader.mjs';
import { runSubtask as defaultRunSubtask } from '../subtask-runner.mjs';

const FRAMEWORK_DIR = dirname(fileURLToPath(import.meta.url));
const SCRIPT_DIR = dirname(dirname(FRAMEWORK_DIR));
const VALID_SUBTASKS = new Set(['metadata', 'team', 'funding', 'audits']);

function freshRunId() {
  return new Date().toISOString().replace(/[-:]/g, '').replace(/\.\d{3}Z$/, 'Z');
}

async function defaultValidate(record, manifestPath) {
  const manifest = await loadManifest(manifestPath);
  const result = await validateRecord(record, manifest);
  return { ok: result.ok, errors: result.errors || [] };
}

export default async function refreshCmd(args, ctx = {}) {
  const outputRoot = ctx.outputRoot || join(SCRIPT_DIR, 'out');
  const manifestPath = ctx.manifestPath || join(SCRIPT_DIR, 'consumers', 'protocol-info', 'manifest.json');
  const runSubtask = ctx.runSubtask || defaultRunSubtask;
  const merge = ctx.merge || mergeR2;
  const validate = ctx.validate || ((rec) => defaultValidate(rec, manifestPath));

  const [slug, subtask] = args;
  if (!slug || !subtask) {
    process.stderr.write('Usage: protocol-info refresh <slug> <subtask>\n');
    process.stderr.write('  <subtask> ∈ {metadata, team, funding, audits}\n');
    return 1;
  }
  if (!VALID_SUBTASKS.has(subtask)) {
    process.stderr.write(`refresh: unknown subtask "${subtask}". Must be one of: ${[...VALID_SUBTASKS].join(', ')}\n`);
    return 1;
  }
  const recordPath = join(outputRoot, slug, 'record.json');
  if (!existsSync(recordPath)) {
    process.stderr.write(`refresh: ${recordPath} does not exist. Run 'crawl' first.\n`);
    return 1;
  }
  const existing = JSON.parse(await readFile(recordPath, 'utf8'));
  const result = await runSubtask({ slug, subtask, existing, manifestPath, outputRoot });
  if (!result || typeof result !== 'object') {
    process.stderr.write(`refresh: subtask returned no result\n`);
    return 1;
  }
  const merged = merge(existing, result);
  const v = await validate(merged);
  if (!v.ok) {
    process.stderr.write(`refresh: post-merge validation failed (${v.errors.length} errors). NOT committing.\n`);
    for (const e of v.errors.slice(0, 5)) {
      process.stderr.write(`  ${e.path || '/'}: ${e.message}\n`);
    }
    return 1;
  }
  await writeFile(recordPath, JSON.stringify(merged, null, 2) + '\n');
  await commit(outputRoot, {
    paths: [`${slug}/`],
    message: `refresh(${slug}): ${subtask}`,
    runId: freshRunId(),
  });
  await buildOutBrowser(outputRoot);
  return 0;
}
```

Read the file first to understand its current API.
